1. Configurare Git și clonare repository

git config --global user.email "loghin.daniel@elev.cihcahul.md"
git clone https://github.com/vfugvugvcft3456/Evaluare-Nr2.git

2. Creare și commit fișier settings.xml

echo "<settings>initial</settings>" > settings.xml
git add settings.xml
git commit -m "Commit file settings.xml"

3. Redenumire fișier și amendare commit

git mv settings.xml configuration.xml
git commit -m "Commit configuration.xml"
git commit --amend -m "Commit file configuration.xml"

4. Push forțat către remote

git push origin main --force

5. Creare și commit fișier page1.html

echo "<html><body>Page1</body></html>" > page1.html
git add page1.html
git commit -m "Add page1.html"
git push origin main

6. Creare branch stage și fișier hosts.xml

git checkout -b stage main
git push -u origin stage
echo "<hosts>localhost</hosts>" > hosts.xml
git add hosts.xml
git commit -m "Commit file hosts.xml"
git push origin stage
git switch main
git merge stage
git push origin main

7. Branch dev → develop și sincronizare

git switch stage
git checkout -b dev
echo "ENV=production" > env.txt
git add env.txt
git commit -m "Commit file env.txt"
git push -u origin dev
git switch stage
git merge dev
git push origin stage
git switch dev
git branch -m develop
git push origin :dev
git push -u origin develop
